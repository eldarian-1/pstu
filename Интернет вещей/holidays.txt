Для Python:
docker build -t holidays .
docker run --name holidays --network host -d holidays
docker run -p 8888:80 --name holidays --network host -d holidays
docker stop holidays
docker rm $(docker ps -qa)
 
docker run --rm --name holidays -p 8000:8000 -t -e TZ=Europe/Moscow holidays

Для NGINX:
docker-compose up
docker-compose down
